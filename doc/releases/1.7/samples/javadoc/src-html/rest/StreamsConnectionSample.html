005package rest;
006
007import java.io.IOException;
008import java.util.List;
009
010import com.ibm.streamsx.rest.InputPort;
011import com.ibm.streamsx.rest.Instance;
012import com.ibm.streamsx.rest.Job;
013import com.ibm.streamsx.rest.Metric;
014import com.ibm.streamsx.rest.Operator;
015import com.ibm.streamsx.rest.OutputPort;
016import com.ibm.streamsx.rest.ProcessingElement;
017import com.ibm.streamsx.rest.RESTException;
018import com.ibm.streamsx.rest.StreamsConnection;
019
020/**
021 * This is the main class to show how to use the StreamsConnection.
022 *
023 * <p>
024 * The example connects to the streams instance and gets a list of resources The
025 * objects printed out include all the fields known in the JSON object per the
026 * REST api however, the classes may not have accessors for all objects
027 * </p>
028 *
029 * <p>
030 * The following arguments are required for the sample:
031 * <ul>
032 * <li>userName</li>
033 * <li>password</li>
034 * <li>rest url for the IBM Streams Instance (streamstool geturl --api)</li>
035 * <li>instance name</li>
036 * </ul>
037 * </p>
038 */
039public class StreamsConnectionSample {
040
041    public static void main(String[] args) throws IOException {
042        String userName = args[0];
043        String authToken = args[1];
044        String url = args[2];
045        String instanceName = args[3];
046
047        /*
048         * Create the connection to the instance indicated
049         */
050        StreamsConnection sClient = StreamsConnection.createInstance(userName, authToken, url);
051
052        /*
053         * This option is only used to by-pass the certificate certification
054         */
055        if (args.length == 5) {
056            if (args[4].equals("true")) {
057                sClient.allowInsecureHosts(true);
058            }
059        }
060
061        try {
062            System.out.println("Instance: ");
063            Instance instance = sClient.getInstance(instanceName);
064
065            /*
066             * From the Instance, get a list of jobs
067             */
068            List<Job> jobs = instance.getJobs();
069            for (Job job : jobs) {
070                System.out.println("Job: " + job.toString());
071                /*
072                 * For each job, get a list of operators
073                 */
074                List<Operator> operators = job.getOperators();
075                for (Operator op : operators) {
076                    System.out.println("Operator: " + op.toString());
077                    List<Metric> metrics = op.getMetrics();
078                    /*
079                     * For each operator, you can get a list of metrics, output
080                     * ports and input ports
081                     */
082                    for (Metric m : metrics) {
083                        System.out.println("Metric: " + m.toString());
084                    }
085                    List<OutputPort> outP = op.getOutputPorts();
086                    for (OutputPort oport : outP) {
087                        System.out.println("Output Port: " + oport.toString());
088                        /*
089                         * For each output port, you can get a list of metrics
090                         */
091                        for (Metric om : oport.getMetrics()) {
092                            System.out.println("Output Port Metric: " + om.toString());
093                        }
094                    }
095                    List<InputPort> inP = op.getInputPorts();
096                    /*
097                     * For each input port, get a list of metrics
098                     */
099                    for (InputPort ip : inP) {
100                        System.out.println("Input Port: " + ip.toString());
101                        for (Metric im : ip.getMetrics()) {
102                            System.out.println("Input Port Metric: " + im.toString());
103                        }
104                    }
105                }
106                /*
107                 * For each job, get a list of processing elements
108                 */
109                for (ProcessingElement pe : job.getPes()) {
110                    System.out.println("ProcessingElement:" + pe.toString());
111                }
112            }
113
114            try {
115                /*
116                 * Get a specific job in the instance
117                 */
118                instance.getJob("99999");
119            } catch (RESTException e) {
120                /*
121                 * This shows what is available in the RESTException should
122                 * something fail
123                 */
124                System.out.println("Status Code: " + e.getStatusCode());
125                System.out.println("Message Id: " + e.getStreamsErrorMessageId());
126                System.out.println("MessageAsJson: " + e.getStreamsErrorMessageAsJson().toString());
127                System.out.println("Message: " + e.getMessage());
128            }
129        } catch (Exception e) {
130            e.printStackTrace();
131        }
132    }
133}
