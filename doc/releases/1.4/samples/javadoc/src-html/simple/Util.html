005package simple;
006
007import java.io.File;
008import java.util.HashMap;
009import java.util.Map;
010import java.util.logging.Level;
011import java.util.logging.Logger;
012
013import com.ibm.streams.operator.logging.TraceLevel;
014import com.ibm.streamsx.topology.context.ContextProperties;
015import com.ibm.streamsx.topology.context.JobProperties;
016import com.ibm.streamsx.topology.context.StreamsContext;
017
018public class Util {
019    private static Logger LOGGER = Logger.getLogger("simple.util");
020    
021    private static Map<String,Level> levelMap = new HashMap<String,Level>();
022    static {
023        levelMap.put("off", TraceLevel.OFF);
024        levelMap.put("error", TraceLevel.ERROR);
025        levelMap.put("warn", TraceLevel.WARN);
026        levelMap.put("info", TraceLevel.INFO);
027        levelMap.put("trace", TraceLevel.TRACE);
028        levelMap.put("debug", TraceLevel.DEBUG);
029    }
030    
031    /**
032     * Look for configuration properties present in <code>args</code> 
033     * and sets the property in the <code>config</code> map.
034     * <p>
035     * An property item in <code>args</code> is required to be in the format:
036     * <pre>
037     * &lt;property-name>=&lt;value>
038     * </pre>
039     * @param args list to scan.  Unrecognized items are silently ignored.
040     * @return configuration map
041     * @see StreamsContext#submit(com.ibm.streamsx.topology.Topology, Map)
042     */
043    public static Map<String,Object> createConfig(String[] args) {
044        String[] cfgItems = {
045                ContextProperties.APP_DIR,  // String
046                ContextProperties.KEEP_ARTIFACTS, // Boolean
047                ContextProperties.TOOLKIT_DIR,  // String
048                ContextProperties.TRACING_LEVEL,  // java.util.logging.Level
049                // TODO ContextProperties.VMARGS,   // List<String>
050                JobProperties.GROUP,  // String
051                JobProperties.NAME,   // String
052                JobProperties.OVERRIDE_RESOURCE_LOAD_PROTECTION, // Boolean
053                JobProperties.DATA_DIRECTORY, // String
054                JobProperties.PRELOAD_APPLICATION_BUNDLES,  // Boolean
055        };
056
057        Map<String,Object> config = new HashMap<String,Object>();
058
059        // handle param=<value>
060        for (String arg : args) {
061            String[] toks = arg.split("=", 2);
062            if (toks.length > 1) {
063                for (String item : cfgItems) {
064                    if (toks[0].equals(item)) {
065                        String valStr = toks[1];
066                        Object value = valStr;
067                        if (item.equals(ContextProperties.TRACING_LEVEL))
068                            value = toTracingLevel(valStr);
069                        else if (item.equals(ContextProperties.KEEP_ARTIFACTS))
070                            value = Boolean.valueOf(valStr);
071                        else if (item.equals(JobProperties.OVERRIDE_RESOURCE_LOAD_PROTECTION))
072                            value = Boolean.valueOf(valStr);
073                        else if (item.equals(JobProperties.PRELOAD_APPLICATION_BUNDLES))
074                            value = Boolean.valueOf(valStr);
075                        
076                        config.put(item, value);
077                        LOGGER.info("Setting config: item="+item+" value="+value+"  [arg="+arg+"]");
078                    }
079                }
080            }
081        }
082        
083        return config;
084    }
085    
086    private static Level toTracingLevel(String str) {
087        Level l = levelMap.get(str.toLowerCase());
088        if (l==null)
089            throw new IllegalArgumentException("Unrecognized tracing level '"+str+"'."
090                    + " Must be one of: "+levelMap.keySet());
091        return l;
092    }
093
094}
