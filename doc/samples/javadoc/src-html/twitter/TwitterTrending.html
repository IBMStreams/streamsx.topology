005package twitter;
006
007import static com.ibm.streamsx.topology.file.FileStreams.directoryWatcher;
008import static com.ibm.streamsx.topology.file.FileStreams.textFileReader;
009
010import java.io.ObjectStreamException;
011import java.util.ArrayList;
012import java.util.regex.Matcher;
013import java.util.regex.Pattern;
014
015import com.ibm.streamsx.topology.TStream;
016import com.ibm.streamsx.topology.Topology;
017import com.ibm.streamsx.topology.context.StreamsContextFactory;
018import com.ibm.streamsx.topology.function7.Function;
019
020/**
021 * Sample twitter trending topology application. This Java application builds a 
022 * topology that reads from a file of tweets, extracts the hashtags from each
023 * line, and uses a window to keep track of the most popular hashtags from the
024 * past 40,000 tweets.
025 * 
026 * <br><br>
027 * 
028 * Although the application reads from a file, in principle it could be attached
029 * to a live data source.
030 * 
031 * <BR>
032 * <P>
033 * If no arguments are provided then the topology is executed in embedded mode,
034 * within this JVM.
035 * <BR>
036 * This may be executed as (from the {@code samples/java/functional directory} )
037 * as:
038 * <UL>
039 * <LI>{@code ant run.helloworld} - Using Apache Ant, this will run in embedded
040 * mode.</LI>
041 * <LI>
042 * {@code java -jar funcsamples.jar:../com.ibm.streamsx.topology/lib/com.ibm.streamsx.topology.jar:$STREAMS_INSTALL/lib/com.ibm.streams.operator.samples.jar simple.TwitterTrending CONTEXT_TYPE DIRECTORY}
043 * <i>CONTEXT_TYPE</i> - Run directly from the command line with a specific
044 * {@code StreamsContext} where <i>CONTEXT_TYPE</i> is one of:
045 * <UL>
046 * <LI>{@code DISTRIBUTED} - Run as an IBM InfoSphere Streams distributed
047 * application.</LI>
048 * <LI>{@code STANDALONE} - Run as an IBM InfoSphere Streams standalone
049 * application.</LI>
050 * <LI>{@code EMBEDDED} - Run embedded within this JVM.</LI>
051 * <LI>{@code BUNDLE} - Create an IBM InfoSphere Streams application bundle.</LI>
052 * <LI>{@code TOOLKIT} - Create an IBM InfoSphere Streams application toolkit.</LI>
053 * </UL>
054 * and <i>DIRECTORY</i> is the location of a directory that contains one or more
055 * text files containing lines of tweets.
056 * </LI>
057 * <LI>
058 * An application execution within your IDE once you set the class path to include the correct jars.</LI>
059 * </UL>
060 * </P>
061 */
062public class TwitterTrending {
063    private static final Pattern TAG_PATTERN = Pattern
064            .compile("(?:^|\\s|[\\p{Punct}&&[^/]])(#[\\p{L}0-9-_]+)");
065
066    @SuppressWarnings("serial")
067    public static void main(String args[]) throws Exception {
068        if(args.length == 0){
069            throw new IllegalArgumentException("Must supply CONTEXT_TYPE and DIRECTORY as arguments");
070        }
071        String contextType = args[0];
072        String directory = args[1];
073
074        // Define the topology
075        Topology topology = new Topology("twitterPipeline");
076
077        // Stream containing file with tweets
078        TStream<String> files = directoryWatcher(topology, directory);
079
080        // Create a stream of lines from each file.
081        TStream<String> lines = textFileReader(files);
082
083        // Extract the hashtags from the string
084        TStream<String> hashtags = lines.multiTransform(
085                new Function<String, Iterable<String>>() {
086
087                    @Override
088                    public Iterable<String> apply(String v1) {
089                        ArrayList<String> tweetHashTags = new ArrayList<String>();
090                        matcher.reset(v1);
091                        while (matcher.find()) {
092                            tweetHashTags.add(matcher.group(1));
093                        }
094                        return tweetHashTags;
095                    }
096
097                    transient Matcher matcher;
098
099                    private Object readResolve() throws ObjectStreamException {
100                        matcher = TAG_PATTERN.matcher("");
101                        return this;
102                    }
103
104                }, String.class);
105
106        // Extract the most frequent hashtags
107        TStream<ListContainer> hashTagMap = hashtags.last(40000).aggregate(
108                new Function<Iterable<String>, ListContainer>() {
109
110                    @Override
111                    public ListContainer apply(Iterable<String> v1) {
112                        Trender tre = new Trender();
113                        for (String s_iter : v1) {
114                            tre.add(s_iter);
115                        }
116                        return tre.getTopTen();
117                    }
118
119                }, ListContainer.class);
120
121        hashTagMap.print();
122
123        StreamsContextFactory.getStreamsContext(contextType).submit(topology);
124    }
125}
