005package twitter;
006
007import java.util.ArrayList;
008import java.util.Collections;
009import java.util.Comparator;
010import java.util.HashMap;
011import java.util.Map.Entry;
012
013/**
014 * A container for a hashmap that keeps the top ten of whatever is in the map.
015 */
016public class Trender {
017    // Overall hashmap for hashtags -- maintains the count for each hashtag
018    private HashMap<String, Integer> hmap = new HashMap<String, Integer>();
019
020    public HashMap<String, Integer> getMap() {
021        return hmap;
022    }
023
024    /*
025     * Adds a hashtag to the table of hashtags, increasing its count.
026     */
027    public void add(String s) {
028        Integer val;
029
030        // Insert the hashtag into the map.
031        if (!hmap.containsKey(s)) {
032            val = 1;
033            hmap.put(s, val);
034        } else {
035            val = hmap.get(s) + 1;
036            hmap.put(s, val);
037        }
038    }
039
040    /**
041     * Get a hashMap element
042     * 
043     * @param s
044     * @return an element of the hashmap
045     */
046    public Integer get(String s) {
047        return hmap.get(s);
048    }
049
050    /**
051     * Gets the top ten most frequent hashtags as a list.
052     * 
053     * @return A ListContainer containing the top ten most frequent hashtags.
054     */
055    public ListContainer getTopTen() {
056        ArrayList<HashTagCount> topTen = new ArrayList<HashTagCount>();
057
058        // String[] topTen= new String[10];
059        Integer minValue = 0;
060        for (Entry<String, Integer> eset : hmap.entrySet()) {
061            if (eset.getValue() < minValue) {
062                continue;
063            }
064            if (topTen.size() > 10) {
065                topTen.remove(0);
066            }
067            HashTagCount htc = new HashTagCount(eset.getKey(), eset.getValue());
068            int idx = Collections.binarySearch(topTen, htc,
069                    new Comparator<HashTagCount>() {
070
071                        @Override
072                        public int compare(HashTagCount o1, HashTagCount o2) {
073                            if (o1.getCount() > o2.getCount())
074                                return 1;
075                            else if (o1.getCount().equals(o2.getCount()))
076                                return 0;
077                            else
078                                return -1;
079                        }
080                    });
081            if (idx < 0)
082                idx = (-1 * idx) - 1;
083            topTen.add(idx, htc);
084            minValue = topTen.get(0).getCount();
085        }
086        return new ListContainer(topTen);
087    }
088
089}
