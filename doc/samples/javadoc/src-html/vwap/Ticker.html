005package vwap;
006
007import java.io.Serializable;
008
009import com.ibm.streams.operator.Tuple;
010import com.ibm.streamsx.topology.tuple.Keyable;
011
012public abstract class Ticker implements Keyable<String>, Serializable {
013
014    /**
015     * 
016     */
017    private static final long serialVersionUID = 1L;
018    private final String ticker;
019
020    Ticker(Tuple tuple) {
021        this(tuple.getString("ticker"));
022    }
023
024    Ticker(String ticker) {
025        this.ticker = ticker;
026    }
027
028    Ticker(Ticker ticker) {
029        this.ticker = ticker.getTicker();
030    }
031
032    @Override
033    public String getKey() {
034        return getTicker();
035    }
036
037    public String getTicker() {
038        return ticker;
039    }
040}
