005package vwap;
006
007import com.ibm.streams.operator.StreamSchema;
008import com.ibm.streams.operator.Type;
009import com.ibm.streamsx.topology.TStream;
010import com.ibm.streamsx.topology.Topology;
011import com.ibm.streamsx.topology.context.StreamsContextFactory;
012import com.ibm.streamsx.topology.function7.Predicate;
013import com.ibm.streamsx.topology.spl.FileSPLStreams;
014import com.ibm.streamsx.topology.spl.FileSPLStreams.Compression;
015import com.ibm.streamsx.topology.spl.SPLStream;
016
017public class Vwap {
018    private static StreamSchema TQRecT = Type.Factory
019            .getStreamSchema("tuple<rstring ticker,rstring date, rstring time, int32 gmtOffset,"
020                    + "rstring ttype, rstring exCntrbID, decimal64 price,"
021                    + "decimal64 volume, decimal64 vwap, rstring buyerID,"
022                    + "decimal64 bidprice, decimal64 bidsize, int32 numbuyers,"
023                    + "rstring sellerID, decimal64 askprice, decimal64 asksize,"
024                    + "int32 numsellers, rstring qualifiers, int32 seqno,"
025                    + "rstring exchtime, decimal64 blockTrd, decimal64 floorTrd,"
026                    + "decimal64 PEratio, decimal64 yield, decimal64 newprice,"
027                    + "decimal64 newvol, int32 newseqno, decimal64 bidimpvol,"
028                    + "decimal64 askimpcol, decimal64 impvol>");
029
030    /**
031     * Run the Vwap topology, printing the bargains to stdout.
032     * @param args
033     */
034    public static void main(String[] args) throws Exception {
035
036        TStream<Bargain> bargains = createVwapTopology();
037
038        bargains.print();
039
040        StreamsContextFactory.getStreamsContext(args[0]).submit(bargains.topology()).get();
041    }
042    
043    /**
044     * Create the Vwap topology, returning the resultant bargains stream.
045     */
046    public static TStream<Bargain> createVwapTopology() {
047        Topology topology = new Topology("Vwap");
048
049        String vwapDataFile = System.getenv("STREAMS_INSTALL")
050                + "/samples/spl/application/Vwap/data/TradesAndQuotes.csv.gz";
051
052        TStream<String> vwapDataFileName = topology.strings(vwapDataFile);
053        SPLStream tradeQuotes = FileSPLStreams.csvCompressedReader(
054                vwapDataFileName, TQRecT, Compression.gzip);
055
056        // Convert the SPLStreams into Stream<T> instances,
057        // unpacking the SPL Tuple into Quote and Trade objects
058        TStream<Trade> trades = Trade.getTrades(tradeQuotes);
059        TStream<Quote> quotes = Quote.getQuotes(tradeQuotes);
060
061        TStream<Bargain> bargains = VwapProcessing.bargains(trades, quotes);
062        
063        return bargains;
064    }
065    
066    public static TStream<Bargain> realBargains(TStream<Bargain> bargains) {
067        // Add a filter for actual bargains
068        return bargains.filter(new Predicate<Bargain>() {
069            private static final long serialVersionUID = 1L;
070
071            @Override
072            public boolean test(Bargain bargain) {
073                return bargain.isBargain();
074            }
075        });
076    }
077}
