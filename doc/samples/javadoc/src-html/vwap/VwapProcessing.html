005package vwap;
006
007import java.util.List;
008
009import com.ibm.streamsx.topology.TStream;
010import com.ibm.streamsx.topology.function.BiFunction;
011import com.ibm.streamsx.topology.function.Function;
012import com.ibm.streamsx.topology.logic.Logic;
013
014public class VwapProcessing {
015
016    @SuppressWarnings("serial")
017    public static TStream<Bargain> bargains(TStream<Trade> trades,
018            TStream<Quote> quotes) {
019
020        TStream<VWapT> vwap = trades.last(4).aggregate(
021                new Function<List<Trade>, VWapT>() {
022
023                    @Override
024                    public VWapT apply(List<Trade> tuples) {
025                        VWapT vwap = null;
026                        for (Trade trade : tuples) {
027                            if (vwap == null)
028                                vwap = new VWapT(trade);
029                            vwap.newTrade(trade);
030                        }
031                        return vwap == null ? null : vwap.complete();
032                    }
033                }, VWapT.class);
034
035        TStream<Bargain> bargainIndex = quotes.join(vwap.last(),
036                Logic.first(new BiFunction<Quote, VWapT, Bargain>() {
037
038                    @Override
039                    public Bargain apply(Quote v1, VWapT v2) {
040                        return new Bargain(v1, v2);
041                    }
042                }), Bargain.class);
043
044        return bargainIndex;
045    }
046}
