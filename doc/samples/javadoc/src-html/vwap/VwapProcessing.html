005package vwap;
006
007import com.ibm.streamsx.topology.TStream;
008import com.ibm.streamsx.topology.function7.BiFunction;
009import com.ibm.streamsx.topology.function7.Function;
010import com.ibm.streamsx.topology.logic.Logic;
011
012public class VwapProcessing {
013
014    public static TStream<Bargain> bargains(TStream<Trade> trades,
015            TStream<Quote> quotes) {
016
017        TStream<VWapT> vwap = trades.last(4).aggregate(
018                new Function<Iterable<Trade>, VWapT>() {
019
020                    @Override
021                    public VWapT apply(Iterable<Trade> tuples) {
022                        VWapT vwap = null;
023                        for (Trade trade : tuples) {
024                            if (vwap == null)
025                                vwap = new VWapT(trade);
026                            vwap.newTrade(trade);
027                        }
028                        return vwap == null ? null : vwap.complete();
029                    }
030                }, VWapT.class);
031
032        TStream<Bargain> bargainIndex = quotes.join(vwap.last(),
033                Logic.first(new BiFunction<Quote, VWapT, Bargain>() {
034
035                    @Override
036                    public Bargain apply(Quote v1, VWapT v2) {
037                        return new Bargain(v1, v2);
038                    }
039                }), Bargain.class);
040
041        return bargainIndex;
042    }
043}
