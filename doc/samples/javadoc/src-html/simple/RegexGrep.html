005package simple;
006
007import java.io.ObjectStreamException;
008import java.util.concurrent.Future;
009import java.util.regex.Matcher;
010import java.util.regex.Pattern;
011
012import com.ibm.streamsx.topology.TStream;
013import com.ibm.streamsx.topology.Topology;
014import com.ibm.streamsx.topology.context.StreamsContextFactory;
015import com.ibm.streamsx.topology.file.FileStreams;
016import com.ibm.streamsx.topology.function7.Predicate;
017
018/**
019 * Sample continuous (streaming) regular expression grep topology application.
020 * This is a variant of the {@link Grep} application that demonstrates
021 * filtering using Java functional programming.
022 * This Java application builds a
023 * simple topology that watches a directory for files, reads each file and
024 * output lines that match a regular expression.
025 * Thus as each file is added to the directory, the application will read
026 * it and output matching lines.
027 * <BR>
028 * The application implements the typical pattern of code that declares a
029 * topology followed by submission of the topology to a Streams context {@code
030 * com.ibm.streamsx.topology.context.StreamsContext}.
031 * <BR>
032 * This demonstrates Java functional programming using an anonymous class.
033 * <P>
034 * <BR>
035 * This may be executed as (from the {@code samples/java/functional directory} )
036 * as:
037 * <UL>
038 * <LI>
039 * {@code java -jar funcsamples.jar:../com.ibm.streamsx.topology/lib/com.ibm.streamsx.topology.jar:$STREAMS_INSTALL/lib/com.ibm.streams.operator.samples.jar simple.RegexGrep} <i>CONTEXT_TYPE</i>
040 * {@code $HOME/books .*Queen.*England.*}
041 * - Run directly from the command line with a specific
042 * {@code StreamsContext} where <i>CONTEXT_TYPE</i> is one of:
043 * <UL>
044 * <LI>{@code DISTRIBUTED} - Run as an IBM InfoSphere Streams distributed
045 * application.</LI>
046 * <LI>{@code STANDALONE} - Run as an IBM InfoSphere Streams standalone
047 * application.</LI>
048 * <LI>{@code EMBEDDED} - Run embedded within this JVM.</LI>
049 * <LI>{@code BUNDLE} - Create an IBM InfoSphere Streams application bundle.</LI>
050 * <LI>{@code TOOLKIT} - Create an IBM InfoSphere Streams application toolkit.</LI>
051 * </UL>
052 * </LI>
053 * <LI>
054 * An application execution within your IDE once you set the class path to include the correct jars.</LI>
055 * </UL>
056 * </P>
057 */
058public class RegexGrep {
059    @SuppressWarnings("serial")
060    public static void main(String[] args) throws Exception {
061        String contextType = args[0];
062        String directory = args[1];
063        final Pattern pattern = Pattern.compile(args[2]);
064
065        // Define the topology
066        Topology topology = new Topology("RegexGrep");
067
068        // All streams with tuples that are Java String objects
069        TStream<String> files = FileStreams.directoryWatcher(topology, directory);
070        TStream<String> lines = FileStreams.textFileReader(files);
071        
072        /*
073         * Functional filter using an anonymous class to define the
074         * filtering logic, in this case execution of a regular
075         * expression against each input String tuple (each line
076         * of the files in the directory).
077         */
078        TStream<String> filtered = lines.filter(new Predicate<String>() {
079
080            @Override
081            public boolean test(String v1) {
082                // Pass the line through if it matches the
083                // regular expression pattern
084                return matcher.reset(v1).matches();
085            }
086
087            // Recreate the matcher (which is not serializable)
088            // when the object is deserialized using readResolve.
089            transient Matcher matcher;
090
091            /*
092             * Since the constructor is no invoked after serialization
093             * we use readResolve as a hook to execute initialization
094             * code, in this case creating the matcher from the
095             * pattern. 
096             * The alternative would be to create it on its first use,
097             * which would require an if statement in the test method.
098             */
099            private Object readResolve() throws ObjectStreamException {
100                matcher = pattern.matcher("");
101                return this;
102            }
103        });
104
105        // For debugging just print out the tuples
106        filtered.print();
107
108        // Execute the topology, just like Grep.
109        Future<?> future = StreamsContextFactory.getStreamsContext(contextType)
110                .submit(topology);
111        Thread.sleep(30000);
112        future.cancel(true);
113    }
114}
