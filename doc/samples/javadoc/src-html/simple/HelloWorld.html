005package simple;
006
007import com.ibm.streamsx.topology.TStream;
008import com.ibm.streamsx.topology.Topology;
009import com.ibm.streamsx.topology.context.StreamsContextFactory;
010
011/**
012 * Sample Hello World topology application. This Java application builds a
013 * simple topology that prints Hello World to standard output. <BR>
014 * The application implements the typical pattern of code that declares a
015 * topology followed by submission of the topology to a Streams context (@code
016 * com.ibm.streamsx.topology.context.StreamsContext}.
017 * <BR>
018 * This demonstrates the mechanics of declaring a topology and executing it.
019 * <P>
020 * If no arguments are provided then the topology is executed in embedded mode,
021 * within this JVM.
022 * <BR>
023 * This may be executed as (from the {@code samples/java/functional directory} )
024 * as:
025 * <UL>
026 * <LI>{@code ant run.helloworld} - Using Apache Ant, this will run in embedded
027 * mode.</LI>
028 * <LI>
029 * {@code java -jar funcsamples.jar:../com.ibm.streamsx.topology/lib/com.ibm.streamsx.topology.jar:$STREAMS_INSTALL/lib/com.ibm.streams.operator.samples.jar simple.HelloWorld}
030 * - Run directly from the command line in embedded mode.
031 * <LI>
032 * {@code java -jar funcsamples.jar:../com.ibm.streamsx.topology/lib/com.ibm.streamsx.topology.jar:$STREAMS_INSTALL/lib/com.ibm.streams.operator.samples.jar simple.HelloWorld}
033 * <i>CONTEXT_TYPE</i> - Run directly from the command line with a specific
034 * {@code StreamsContext} where <i>CONTEXT_TYPE</i> is one of:
035 * <UL>
036 * <LI>{@code DISTRIBUTED} - Run as an IBM InfoSphere Streams distributed
037 * application.</LI>
038 * <LI>{@code STANDALONE} - Run as an IBM InfoSphere Streams standalone
039 * application.</LI>
040 * <LI>{@code EMBEDDED} - Run embedded within this JVM.</LI>
041 * <LI>{@code BUNDLE} - Create an IBM InfoSphere Streams application bundle.</LI>
042 * <LI>{@code TOOLKIT} - Create an IBM InfoSphere Streams application toolkit.</LI>
043 * </UL>
044 * </LI>
045 * <LI>
046 * An application execution within your IDE once you set the class path to include the correct jars.</LI>
047 * </UL>
048 * </P>
049 */
050public class HelloWorld {
051
052    /**
053     * Sample Hello World topology application.
054     * This Java application builds a simple topology
055     * that prints Hello World to standard output.
056     * <BR>
057     * The application implements the typical pattern
058     * of code that declares a topology followed by
059     * submission of the topology to a Streams context
060     * (@code com.ibm.streamsx.topology.context.StreamsContext}.
061
062     * @param args
063     * @throws Exception
064     */
065    public static void main(String[] args) throws Exception {
066
067        /*
068         * Create the container for the topology that will
069         * hold the streams of tuples.
070         */
071        Topology topology = new Topology("HelloWorld");
072
073        /*
074         * Declare a source stream (hw) with String tuples containing two tuples,
075         * "Hello" and "World!".
076         */
077        TStream<String> hw = topology.strings("Hello", "World!");
078        
079        /*
080         * Sink hw by printing each of its tuples to System.out.
081         */
082        hw.print();
083        
084        /*
085         * At this point the topology is declared with a single
086         * stream that is printed to System.out.
087         */
088
089        /*
090         * Now execute the topology by submitting to a StreamsContext.
091         * If no argument is provided then the topology is executed
092         * within this JVM (StreamsContext.Type.EMBEDDED).
093         * Otherwise the first and only argument is taken as the
094         * String representation of the 
095         */
096        if (args.length == 0)
097            StreamsContextFactory.getEmbedded().submit(topology).get();
098        else
099            StreamsContextFactory.getStreamsContext(args[0]).submit(topology)
100                    .get();
101    }
102}
