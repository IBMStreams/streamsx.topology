005package simple;
006
007import java.util.concurrent.Future;
008
009import com.ibm.streamsx.topology.TStream;
010import com.ibm.streamsx.topology.Topology;
011import com.ibm.streamsx.topology.context.StreamsContextFactory;
012import com.ibm.streamsx.topology.file.FileStreams;
013import com.ibm.streamsx.topology.streams.StringStreams;
014
015/**
016 * Sample continuous (streaming) grep topology application. This Java application builds a
017 * simple topology that watches a directory for files, reads each file and
018 * output lines that contain the search term.
019 * Thus as each file is added to the directory, the application will read
020 * it and output matching lines.
021 * <BR>
022 * The application implements the typical pattern of code that declares a
023 * topology followed by submission of the topology to a Streams context {@code
024 * com.ibm.streamsx.topology.context.StreamsContext}.
025 * <BR>
026 * This demonstrates the a continuous application and use of
027 * utility classes that produce streams.
028 * <P>
029 * <BR>
030 * This may be executed as (from the {@code samples/java/functional directory} )
031 * as:
032 * <UL>
033 * <LI>
034 * {@code java -jar funcsamples.jar:../com.ibm.streamsx.topology/lib/com.ibm.streamsx.topology.jar:$STREAMS_INSTALL/lib/com.ibm.streams.operator.samples.jar simple.Grep} <i>CONTEXT_TYPE</i> {@code $HOME/books Elizabeth}
035 * - Run directly from the command line with a specific
036 * {@code StreamsContext} where <i>CONTEXT_TYPE</i> is one of:
037 * <UL>
038 * <LI>{@code DISTRIBUTED} - Run as an IBM InfoSphere Streams distributed
039 * application.</LI>
040 * <LI>{@code STANDALONE} - Run as an IBM InfoSphere Streams standalone
041 * application.</LI>
042 * <LI>{@code EMBEDDED} - Run embedded within this JVM.</LI>
043 * <LI>{@code BUNDLE} - Create an IBM InfoSphere Streams application bundle.</LI>
044 * <LI>{@code TOOLKIT} - Create an IBM InfoSphere Streams application toolkit.</LI>
045 * </UL>
046 * </LI>
047 * <LI>
048 * An application execution within your IDE once you set the class path to include the correct jars.</LI>
049 * </UL>
050 * </P>
051 */
052public class Grep {
053    
054    /**
055     * Sample continuous (streaming) grep topology application. This Java
056     * application builds a simple topology that watches a directory for
057     * files, reads each file and output lines that contain the search term.
058     * Thus as each file is added to the directory, the application will read
059     * it and output matching lines.
060     * <BR>
061     * The application implements the typical pattern of code that declares a
062     * topology followed by submission of the topology to a Streams context {@code
063     * com.ibm.streamsx.topology.context.StreamsContext}.
064     * <P>
065     * Three arguments are required:
066     * <UL>
067     * <LI>{@code contextType} - The type of the context to execute the topology in, e.g. {@code EMBEDDED or STANDALONE}.</LI>
068     * <LI>{@code directory} - Directory to watch for files.</LI>
069     * <LI>{@code term} - Search term, if any line in a file contains {@code term} then it will be printed.
070     * </UL>
071     * For example (classpath omitted for brevity):
072     * <BR>
073     * {@code java simple.Grep EMBEDDED $HOME/books Elizabeth}
074     */
075    public static void main(String[] args) throws Exception {
076        String contextType = args[0];
077        String directory = args[1];
078        String term = args[2];
079
080        Topology topology = new Topology("Grep");
081
082        /*
083         * Use the file stream utility class com.ibm.streamsx.topology.file.FileStreams
084         * to declare a stream that will contain file names from the specified directory.
085         * As each new file is created in directory its absolute file path will
086         * appear on fileNames.
087         */
088        TStream<String> filePaths = FileStreams.directoryWatcher(topology, directory);
089        
090        /* 
091         * Use the file stream utility class com.ibm.streamsx.topology.file.FileStreams
092         * to declare a stream that will contain the contents of the files.
093         * FileStreams.textFileReader creates a function that for each input
094         * file pat, opens the file and reads its contents as a text file,
095         * producing a tuple for each line of the file. The tuple contains
096         * the contents of the line, as a String.
097         */
098        TStream<String> lines = FileStreams.textFileReader(filePaths);
099        
100        /*
101         * Use the string stream utility class com.ibm.streamsx.topology.streams.StringStreams
102         * to filter out non-matching lines. StringStreams.contains creates a functional
103         * Predicate that will be executed for each tuple on lines, that is each line
104         * read from a file.
105         */
106        TStream<String> matching = StringStreams.contains(lines, term);
107        
108        /*
109         * And print the matching lines to standard out.
110         */
111        matching.print();
112
113        /*
114         * Execute the topology, since FileStreams.directoryWatcher declares
115         * a stream that lasts forever, that it is is always watching the directory,
116         * then when submit() returns the application is not complete.
117         * In fact it will never complete.
118         */
119        Future<?> future = StreamsContextFactory.getStreamsContext(contextType)
120                .submit(topology);
121        
122        /*
123         * Let the application run for thirty seconds (30,000ms)
124         * and then cancel it.
125         */
126        Thread.sleep(30 * 1000);
127        future.cancel(true);
128    }
129}
